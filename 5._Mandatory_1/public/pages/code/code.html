function getRandomColor() {
    let hex = (Math.random() * 0xfffff * 1000000).toString(16);
    return '#' + hex.slice(0, 6);
};

function getRandomInteger(min, max) {
    return Math.floor((Math.random() * (max - min)) + min);
};
